2222 /*
2223  * IMPORTANT NOTICE!
2224  *
2225  * When sending with websocket protocol
2226  *
2227  * LWS_WRITE_TEXT,
2228  * LWS_WRITE_BINARY,
2229  * LWS_WRITE_CONTINUATION,
2230  * LWS_WRITE_PING,
2231  * LWS_WRITE_PONG
2232  *
2233  * the send buffer has to have LWS_PRE bytes valid BEFORE
2234  * the buffer pointer you pass to lws_write().
2235  *
2236  * This allows us to add protocol info before and after the data, and send as
2237  * one packet on the network without payload copying, for maximum efficiency.
2238  *
2239  * So for example you need this kind of code to use lws_write with a
2240  * 128-byte payload
2241  *
2242  * char buf[LWS_PRE + 128];
2243  *
2244  * // fill your part of the buffer... for example here it's all zeros
2245  * memset(&buf[LWS_PRE], 0, 128);
2246  *
2247  * lws_write(wsi, &buf[LWS_PRE], 128, LWS_WRITE_TEXT);
2248  *
2249  * When sending HTTP, with
2250  *
2251  * LWS_WRITE_HTTP,
2252  * LWS_WRITE_HTTP_HEADERS
2253  * LWS_WRITE_HTTP_FINAL
2254  *
2255  * there is no protocol data prepended, and don't need to take care about the
2256  * LWS_PRE bytes valid before the buffer pointer.
2257  *
2258  * LWS_PRE is at least the frame nonce + 2 header + 8 length
2259  * LWS_SEND_BUFFER_POST_PADDING is deprecated, it's now 0 and can be left off.
2260  * The example apps no longer use it.
2261  *
2262  * Pad LWS_PRE to the CPU word size, so that word references
2263  * to the address immediately after the padding won't cause an unaligned access
2264  * error. Sometimes for performance reasons the recommended padding is even
2265  * larger than sizeof(void *).
2266  */
